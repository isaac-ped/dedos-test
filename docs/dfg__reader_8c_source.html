33 static int fix_num_threads(struct dedos_dfg *dfg) {
34  for (int i=0; i<dfg->n_runtimes; i++) {
35  struct dfg_runtime *rt = dfg->runtimes[i];
36 
37  int n_pinned = rt->n_pinned_threads;
38  int n_unpinned = rt->n_unpinned_threads;
39  int total = n_pinned + n_unpinned;
40  for (int t=0; t < total; t++) {
41  struct dfg_thread *thread = rt->threads[t];
42  if (thread->mode == PINNED_THREAD) {
43  n_pinned--;
44  } else if (thread->mode == UNPINNED_THREAD) {
45  n_unpinned--;
46  } else {
47  if (n_pinned > 0) {
48  thread->mode = PINNED_THREAD;
49  log(LOG_DFG_READER, "Setting thread %d with no MSUs to pinned", t);
50  n_pinned--;
51  } else if (n_unpinned > 0) {
52  thread->mode = UNPINNED_THREAD;
53  log(LOG_DFG_READER, "Setting thread %d with no MSUs to unpinned", t);
54  n_unpinned--;
55  } else {
56  log_error("Cannot determine pinned/unpinned status of threads "
57  "based on number of blocking/nonblock MSUs");
58  return -1;
59  }
60  }
61  }
62  if (n_pinned < 0 || n_unpinned < 0) {
63  log_error("More pinned/unpinned threads specified by MSUs than by DFG on rt %d: "
64  "%d more pinned, %d more unpinned", rt->id, -1 * n_pinned, -1 * n_unpinned);
65  return -1;
66  }
67  }
68  return 0;
69 }
