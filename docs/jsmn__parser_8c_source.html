237 static int retry_saved(struct json_state **saved_states, char *j){
238  struct json_state *prev = NULL;
239  struct json_state *saved = *saved_states;
240 
241  int success = -1;
242  char *key = NULL;
243  int n_retries = 0;
244  while (saved != NULL){
245  jsmntok_t *tok = saved->tok;
246  key = tok_to_str(tok, j);
247  log(LOG_JSMN_PARSING, "Retrying key: %s", key);
248 
249  jsmn_parsing_fn parser = get_parse_fn(key, saved->parent_type);
250  tok += 1;
251  int rtn = parser(&tok, j, saved, saved_states);
252 
253  if (rtn < 0){
254  log_error("Error retrying key %s", key);
255  return -1;
256  } else if (rtn == 0){
257  log(LOG_JSMN_PARSING, "Successfully parsed retried key '%s'", key);
258  success = 0;
259  if (*saved_states == saved)
260  *saved_states = saved->next;
261  if (prev != NULL) {
262  prev->next = saved->next;
263  }
264  struct json_state *tmp = saved;
265  saved = saved->next;
266  free(tmp);
267  } else if (rtn == 1){
268  log(LOG_JSMN_PARSING, "Will try again on the next pass...");
269  prev = saved;
270  saved = saved->next;
271  } else {
272  log_error("Unknown return code: %d", rtn);
273  return -1;
274  }
275  if (n_retries++ > MAX_RETRIES) {
276  log_error("Retried too many times. Something is wrong");
277  return -1;
278  }
279  }
280 
281  if (success == -1 && key != NULL) {
282  log_error("Failed while reprocessing key '%s' for missing dependencies", key);
283  log_error("Value for improper key %s", key + (strlen(key) + 1));
284  }
285  return success;
286 }
