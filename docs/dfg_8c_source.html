487  uint32_t new_key) {
488  // This is all so the keys remain in ascending order
489 
490  // Find where the endpoint used to be
491  int old;
492  for (old=0; old<route->n_endpoints && route->endpoints[old] != ep; old++);
493  if (old == route->n_endpoints) {
494  log_error("Could not find endpoint in route");
495  return -1;
496  }
497  // Find where it should now be
498  int new;
499  for (new=0; new<route->n_endpoints && route->endpoints[new]->key < new_key; new++);
500 
501  // Which direction are we moving the in-between endpoints
502  int delta = new > old ? 1 : -1;
503  // If we're deleting from below the new location, subtract one to the new location
504  if (delta == 1) {
505  new -= 1;
506  }
507 
508  // Iterate from the old to the new location, moving the endpoints in the other direction
509  for (int i=old; delta * i < delta * new; i += delta) {
510  route->endpoints[i] = route->endpoints[i + delta];
511  }
512 
513
