695 int init_stat_item(enum stat_id stat_id, unsigned int item_id) {
696  CHECK_INITIALIZATION;
697 
698  struct stat_type *type = &stat_types[stat_id];
699  if (!type->enabled) {
700  log(LOG_STATS, "Skipping initialization of type %s item %d",
701  type->label, item_id);
702  return 0;
703  }
704  if (wrlock_type(type) != 0) {
705  return -1;
706  }
707  if (type->id_indices[item_id] != -1) {
708  log_error("Item ID %u already assigned index %d for stat %s",
709  item_id, type->id_indices[item_id], type->label);
710  return -1;
711  }
712  int index;
713  for (index=0; index<type->max_items; index++) {
714  if (type->items[index].id == -1) {
715  break;
716  }
717  }
718  if (index == type->max_items) {
719  type->max_items++;
720  type->items = realloc(type->items, sizeof(*type->items) * (type->max_items));
721  realloc_stat_samples(stat_id, type->max_items);
722 
723  if ( type->items == NULL ) {
724  log_error("Error reallocating stat item");
725  return -1;
726  }
727  }
728  log(LOG_STATS, "Item %u has index %d", item_id, index);
729  struct stat_item *item = &type->items[index];
730 
731  if (pthread_mutex_init(&item->mutex,NULL)) {
732  log_error("Error initializing item %u lock", item_id);
733  return -1;
734  }
735 
736  item->id = item_id;
737  item->write_index = 0;
738  item->rolled_over = false;
739  item->stats = calloc(type->max_stats, sizeof(*item->stats));
740 
741  if (item->stats == NULL) {
742  log_error("Error calloc'ing item statistics");
743  return -1;
744  }
745 
746  type->id_indices[item_id] = index;
747 
748  type->num_items++;
749 
750  if (unlock_type(type) != 0) {
751  return -1;
752  }
753 
754  log(LOG_STAT_INITS, "Initialized stat item %s.%d (idx: %d)",
755  stat_types[stat_id].label, item_id, index);
756  return 0;
757 }

759 int init_statistics() {
760  if ( stats_initialized ) {
761  log_error("Statistics have already been initialized");
762  return -1;
763  }
764 
765  clock_gettime(CLOCK_ID, &stats_start_time);
766 
767  for (int i = 0; i < N_STAT_TYPES; i++) {
768  struct stat_type *type = &stat_types[i];
769  if ( type->id != i ) {
770  log_error("Stat type %s at incorrect index (%d, not %d)! Disabling!!",
771  type->label, i, type->id);
772  type->enabled = 0;
773  }
774 
775  if (type->enabled) {
776  if (pthread_rwlock_init(&type->lock, NULL) != 0) {
777  log_warn("Error initializing lock for stat %s. Disabling!", type->label);
778  type->enabled = 0;
779  }
780  type->num_items = 0;
781  type->items = NULL;
782  for (int j = 0; j < MAX_STAT_ITEM_ID; j++) {
783  type->id_indices[j] = -1;
784  }
785  log(LOG_STATS, "Initialized stat type %s (%d ids)", type->label, MAX_STAT_ITEM_ID);
786  }
787  }
788 
789 #if DUMP_STATS
790  log_info("Statistics initialized for eventual dump to log file");
791 #else
792  log_info("Statistics initialized. Will **NOT** be dumped to log file");
793 #endif
794  stats_initialized = true;
795  return 0;
796 }
