668  } else {

691 int schedule_msu(struct dfg_msu *msu, struct dfg_runtime *rt, struct dfg_msu **new_msus) {
692  int ret;
693  // First of all, find a thread on the runtime
694  ret = place_on_runtime(rt, msu);
695  if (ret == -1) {
696  log(SCHEDULING, "Could not spawn a new worker thread on runtime %d", rt->id);
697  return -1;
698  }
699 
700  struct dedos_dfg *dfg = get_dfg();
701 
702 
703  //Add the new MSU to the list of new additions
704  memcpy(new_msus, &msu, sizeof(struct dfg_msu *));
705 
706  //Handle dependencies before wiring
707  int l;
708  int skipped = 0;
709  dfg->msus[dfg->n_msus] = msu;
710  dfg->n_msus++;
711  msu->type->instances[msu->type->n_instances] = msu;
712  msu->type->n_instances++;
713  for (l = 0; l < msu->type->n_dependencies; ++l) {
714  struct dfg_msu_type *dep_type = msu->type->dependencies[l]->type;
715  if (msu->type->dependencies[l]->locality == MSU_IS_LOCAL) {
716  struct dfg_msu *existing = msu_type_on_runtime(msu->scheduling.runtime, dep_type);
717  if (existing== NULL) {
718  if (dep_type->n_instances == 0) {
719  log_error("No instances to spawn from!");
720  return -1;
721  }
722  struct dfg_msu *template = dep_type->instances[0];
723  //spawn missing local dependency
724  struct dfg_msu *dep = copy_dfg_msu(template);
725 
726  if (dep == NULL) {
727  log_error("Could not allocate memory for missing dependency");
728  return -1;
729  }
730 
731 
732  prepare_clone(dep);
733  dep->type = msu->type->dependencies[l]->type;
734 
735  ret = schedule_msu(dep, rt, new_msus + l + 1 - skipped);
736  if (ret == -1) {
737  // FIXME: This doesn't properly remove the original MSUs
738  log_info("Could not schedule dependency %d on runtime %d",
739  dep->id, rt->id);
740  dfg->n_msus--;
741  free(dep);
742  return -1;
743  } else {
744  debug("Scheduled dependency %d on runtime %d (l=%d)" , dep->msu_id, rt->id, l);
745  }
746  } else {
747  log_debug("Already has dependency %d", dep_type->id);
748  skipped++;
749  }
750  } else {
751  log_warn("non-local dependency %d", dep_type->id);
752  skipped++;
753  //TODO: check for remote dependency
754  }
755  }
756 
757  log_debug("Processed %d dependencies", l);
758  return 0;
759 }
