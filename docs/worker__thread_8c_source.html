282 int enqueue_worker_timeout(struct worker_thread *thread, struct timespec *interval) {
283  struct timeout_list *tlist = calloc(1, sizeof(*tlist));
284  clock_gettime(CLOCK_REALTIME, &tlist->time);
285  tlist->time.tv_sec += interval->tv_sec;
286  tlist->time.tv_nsec += interval->tv_nsec;
287  if (tlist->time.tv_nsec > 1e9) {
288  tlist->time.tv_nsec -= 1e9;
289  tlist->time.tv_sec += 1;
290  }
291  if (thread->timeouts == NULL) {
292  thread->timeouts = tlist;
293  log(LOG_WORKER_THREAD, "Enqueued timeout to head of queue");
294  return 0;
295  }
296 
297  double diff = timediff_s(&tlist->time, &thread->timeouts->time);
298  if (diff > 0) {
299  tlist->next = thread->timeouts;
300  thread->timeouts = tlist;
301  log(LOG_WORKER_THREAD, "Enqueued timeout to queue");
302  return 0;
303  }
304 
305  struct timeout_list *last_timeout = thread->timeouts;
306  while (last_timeout->next != NULL) {
307  struct timespec *next_timeout = &last_timeout->next->time;
308  diff = timediff_s(&tlist->time, next_timeout);
309  if (diff > 0) {
310  tlist->next = last_timeout->next;
311  last_timeout->next = tlist;
312  log(LOG_WORKER_THREAD, "Enqueued timeout to queue");
313  return 0;
314  }
315  last_timeout = last_timeout->next;
316  }
317  last_timeout->next = tlist;
318  tlist->next = NULL;
319  log(LOG_WORKER_THREAD, "Enqueued timeout to queue");
320  return 0;
321 }

326 static int worker_thread_loop(struct dedos_thread *thread, void *v_worker_thread) {
327  log_info("Starting worker thread loop %d (%s)",
328  thread->id, thread->mode == PINNED_THREAD ? "pinned" : "unpinned");
329 
330  struct worker_thread *self = v_worker_thread;
331 
332  while (!dedos_thread_should_exit(thread)) {
333  // TODO: Get context switches
334  if (thread_wait(thread, next_timeout(self)) != 0) {
335  log_error("Error waiting on thread semaphore");
336  continue;
337  }
338  for (int i=0; i<self->n_msus; i++) {
339  log(LOG_MSU_DEQUEUES, "Attempting to dequeue from msu %d (thread %d)",
340  self->msus[i]->id, thread->id);
341  msu_dequeue(self->msus[i]);
342  }
343  // FIXME: Protect read of num_msgs through mutex
344  int num_msgs = thread->queue.num_msgs;
345  for (int i=0; i<num_msgs; i++) {
346  struct thread_msg *msg = dequeue_thread_msg(&thread->queue);
347  if (msg == NULL) {
348  log_error("Could not read message though queue is not empty!");
349  continue;
350  }
351  log(LOG_THREAD_MESSAGES,"Dequeued thread message on thread %d",
352  thread->id);
353  if (process_worker_thread_msg(self, msg) != 0) {
354  log_error("Error processing worker thread message");
355  }
356  free(msg);
357  }
358  }
359  log_info("Leaving thread %d", thread->id);
360  return 0;
361 }
