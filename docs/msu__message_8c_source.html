141  if (size < sizeof(struct msu_msg_hdr)) {

143  return NULL;
144  }
145  log(LOG_MSU_MSG_READ, "Reading header from %d", fd);
146  struct msu_msg *msg = malloc(sizeof(*msg));
147  if (msg == NULL) {

149  return NULL;

152  log_error("Error reading msu message header");

167  if (read_payload(fd, data_size, data) != 0) {
168  log_perror("Error reading msu msg payload of size %d", (int)data_size);
169  free(msg);
170  free(data);
171  return NULL;
172  }
173  log(LOG_MSU_MSG_DESERIALIZE, "Deserialized MSU message of size %d", (int)data_size);
174  }

178  free(data);
