243 static int process_runtime_message_hdr(struct inter_runtime_msg_hdr *hdr, int fd) {
244  int rtn;
245  switch (hdr->type) {

247  rtn = process_fwd_to_msu_message(hdr->payload_size, hdr->target, fd);
248  if (rtn < 0) {

250  return 1;

252  return 0;

255  if (rtn < 0) {

257  return 1;
258  }
259  return 0;
260  default:
261  log_error("Received unknown message type from fd %d: %d", fd, hdr->type);
262  // Returning 1 because a return of -1 will make the epoll loop exit
263  return 1;
264  }
