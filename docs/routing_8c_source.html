281  uint32_t key, struct msu_endpoint *endpoint) {
282  read_lock(table);
283 
284  unsigned int shortest_queue = MAX_MSU_Q_SIZE;
285  int n_shortest = 0;
286  struct msu_endpoint *endpoints[table->n_endpoints];
287 
288  for (int i=0; i<table->n_endpoints; i++) {
289  if (table->endpoints[i].locality == MSU_IS_REMOTE) {
290  continue;
291  }
292  int length = table->endpoints[i].queue->num_msgs;
293  if (length < shortest_queue) {
294  endpoints[0] = &table->endpoints[i];
295  shortest_queue = length;
296  n_shortest = 1;
297  } else if (length == shortest_queue) {
298  endpoints[n_shortest] = &table->endpoints[i];

301  }
302 
303  if (n_shortest == 0) {
304  unlock(table);
305  return -1;
306  }
307
